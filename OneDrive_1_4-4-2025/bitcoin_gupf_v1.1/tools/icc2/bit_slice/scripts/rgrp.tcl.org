proc group_row {name width} {
    create_rp_group group_${name} -design arm2_top -columns 1 -rows $width
    for {set i 0} {$i < $width} {incr i} {
        add_to_rp_group -leaf [get_cells -hier ${name}_reg_${i}_] -col 0 -row $i arm2_top::group_${name}
        echo "${name}_reg_${i}_"
        echo "group_${name}"
    }
}
